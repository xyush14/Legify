# ---- Subscription / entitlements layer
# pyjwt[crypto] adds the `cryptography` backend so we can verify ES256/RS256

# httpx is the REST client for Supabase service-role calls.
pyjwt[crypto]>=2.8
httpx>=0.27

# Gemini fallback for Anthropic rate-limit / overload. The wrapper in
# headnote/llm/router.py catches anthropic.RateLimitError + 5xx + connection
# errors and retries the same call via Gemini 2.0 Flash (free tier).
# Optional: without GEMINI_API_KEY set the wrapper is a no-op and any
# Anthropic error bubbles up unchanged — identical to the pre-fallback
# behaviour, so this doesn't break existing deploys.
google-genai>=0.3

# Groq is the primary OCR vision backend (free tier, llama-4-scout) for
# FIR photo extraction in /api/draft/ocr-fir. Requires GROQ_API_KEY in env.
# Without the key the OCR endpoint falls back to Anthropic Sonnet vision.
groq>=1.0

# DeepSeek-Reasoner / DeepSeek-Chat (LLM_PROVIDER=deepseek path). DeepSeek's
# API is OpenAI-compatible, so we use the openai SDK pointed at
# api.deepseek.com. Without this dependency, every DeepSeek call silently
# falls through to Groq — burning latency on retries and never billing
# DeepSeek. Critical for the LLM fallback chain.
openai>=1.40

# Resend — transactional email provider for welcome emails and (later)
# payment receipts, password reset confirmations, etc. Requires
# RESEND_API_KEY and a verified domain (headnote.in) with SPF/DKIM/DMARC.
# Without the key the email module no-ops with a loud warning so dev/local
# isn't blocked. See headnote/email/welcome.py.
resend>=2.0

# ---- IK API client + HTML parsing
requests>=2.31
beautifulsoup4>=4.12

# ---- Hindi translation (free; no API key, no LLM cost)
deep-translator>=1.11

# ---- Local paragraph embeddings + cosine search
# fastembed is ~80MB plus the bge model (~80MB more, fetched at first use).
# Without these, the retrieval pipeline degrades to keyword-only.
fastembed>=0.3
numpy>=1.24

# ---- HuggingFace corpus harvest (datasets + tqdm)
# Previously kept in requirements-harvest.txt and pip-installed on demand,
# but Railway sometimes restarts the container mid-install, killing the
# admin harvest job. Moving these into the base image (~250MB cost) so the
# harvest endpoint can run synchronously without any pip phase.
datasets>=2.14
tqdm>=4.66

# ---- PDF rasterization for OCR.
# Groq's vision model (our primary OCR backend) accepts IMAGES ONLY — it
# returns "invalid image data" for PDF uploads. Anthropic can read PDFs but
# is out of credits in production. So when a lawyer uploads a PDF (e.g. a
# Sessions Court order, or an FIR scan), we rasterize each page to a PNG
# with PyMuPDF and send those images to Groq. PyMuPDF bundles its own
# renderer (no system poppler needed) and ships manylinux wheels — installs
# cleanly on Railway with no extra system packages.
PyMuPDF>=1.24

# ---- Server-side PDF export (Print / Download / WhatsApp).
# WeasyPrint renders the drafted document (HTML+CSS) to a REAL, selectable-
# text PDF with correct Devanagari shaping via Pango — replacing the old
# html2canvas client path that rasterized the doc to a broken image (mangled
# conjuncts, busted page breaks). Needs system libs (Pango + Noto/Liberation
# fonts) baked into the Docker image; see Dockerfile. Pin >=60 for stable
# flexbox + CSS-grid handling of the bilingual draft layout.
weasyprint>=60

# ---- Native .docx drafting (Option A): fill Vishnu ji's real filed documents
# (Kruti Dev 010, exact court formatting) as templates instead of LLM-generating
# the document. docxtpl = Jinja tags inside Word; pulls in python-docx + jinja2.
# Field values are converted Unicode->Kruti Dev (headnote/drafter/krutidev.py)
# before injection so the whole doc renders in one consistent legacy font.
docxtpl>=0.18

# ---- Word/Excel attachments. Lawyers routinely have drafts and cause-lists
# as .docx/.xlsx; these are already-digital text, so we extract them directly
# (no vision OCR) wherever we accept PDFs/images. python-docx already ships via
# docxtpl above; openpyxl reads .xlsx sheets (pure-Python, no system deps).
openpyxl>=3.1

# ---- HEIC/HEIF reader so we can OCR iPhone photos. The default iOS photo
# format isn't accepted by Groq/Anthropic vision; Pillow + pillow-heif
# decodes it and we re-encode as JPEG before sending to the OCR provider.
# Pillow comes via fastapi → starlette chain already; explicit pin is fine.
pillow-heif>=0.16; python_version >= "3.9"

# ---- Py 3.9 backport for Pydantic v2 evaluation of `str | None` style hints.
# No-op on 3.10+. Render uses 3.11; safe to keep.
eval_type_backport>=0.2; python_version < "3.10"
